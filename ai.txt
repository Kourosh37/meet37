func Load() *Config {
	turnPort, _ := strconv.Atoi(getEnv("TURN_PORT", "3478"))
	sfuThreshold, _ := strconv.Atoi(getEnv("SFU_FALLBACK_THRESHOLD_KBPS", "1500"))

	return &Config{
		Port:                     getEnv("PORT", "8080"),
		Environment:              getEnv("ENV", "production"),
		AdminUsername:            getEnv("ADMIN_USERNAME", "admin"),
		AdminPassword:            getEnv("ADMIN_PASSWORD", "changeme"),
		JWTSecret:                getEnv("JWT_SECRET", "supersecretkey"),

		TURNSecret:               getEnv("TURN_SECRET", "turnsecret"),

		SFUFallbackThresholdKbps: sfuThreshold,

package db

import (
	"database/sql"
	"fmt"

	_ "github.com/mattn/go-sqlite3"
)

// DB wraps sql.DB with helper methods.
type DB struct {
	*sql.DB
}

// Open opens the SQLite database and applies WAL mode + optimizations.
func Open(path string) (*DB, error) {
	// Use WAL mode for concurrent reads, busy_timeout to avoid SQLITE_BUSY
	dsn := fmt.Sprintf(
		"file:%s?_journal_mode=WAL&_busy_timeout=5000&_synchronous=NORMAL&_cache_size=-64000&_foreign_keys=on",
		path,
	)
	sqlDB, err := sql.Open("sqlite3", dsn)
	if err != nil {
		return nil, fmt.Errorf("open sqlite: %w", err)
	}

	// SQLite is single-writer; limit write concurrency to avoid contention
	sqlDB.SetMaxOpenConns(1)
	sqlDB.SetMaxIdleConns(1)

	db := &DB{sqlDB}
	if err := db.migrate(); err != nil {
		return nil, fmt.Errorf("migrate: %w", err)
	}
	return db, nil
}

// migrate creates all tables if they don't exist.
func (db *DB) migrate() error {
	schema := `
	-- App-wide settings (single row)
	CREATE TABLE IF NOT EXISTS settings (
		id      INTEGER PRIMARY KEY CHECK (id = 1),
		app_mode TEXT NOT NULL DEFAULT 'public' -- 'public' or 'private'
	);

	-- Insert default settings row if not exists
	INSERT OR IGNORE INTO settings (id, app_mode) VALUES (1, 'public');

	-- Registered users (only relevant in private mode)
	CREATE TABLE IF NOT EXISTS users (
		id         TEXT PRIMARY KEY,
		username   TEXT UNIQUE NOT NULL,
		password   TEXT NOT NULL, -- bcrypt hash
		created_at INTEGER NOT NULL
	);

	-- Meeting rooms
	CREATE TABLE IF NOT EXISTS rooms (
		id          TEXT PRIMARY KEY,
		name        TEXT NOT NULL,
		host_id     TEXT NOT NULL,   -- user id or 'admin'
		is_locked   INTEGER NOT NULL DEFAULT 0,
		password    TEXT,            -- optional room password (bcrypt)
		max_peers   INTEGER NOT NULL DEFAULT 50,
		created_at  INTEGER NOT NULL,
		expires_at  INTEGER          -- NULL = no expiry
	);

	-- Room participants snapshot (for analytics/admin view)
	CREATE TABLE IF NOT EXISTS room_events (
		id        INTEGER PRIMARY KEY AUTOINCREMENT,
		room_id   TEXT NOT NULL,
		user_id   TEXT,
		event     TEXT NOT NULL, -- 'join' | 'leave'
		ts        INTEGER NOT NULL
	);

	CREATE INDEX IF NOT EXISTS idx_room_events_room ON room_events(room_id);
	CREATE INDEX IF NOT EXISTS idx_rooms_host ON rooms(host_id);
	`
	_, err := db.Exec(schema)
	return err
}

go
package signaling

import (
	"encoding/json"
	"net/http"
	"sync"
	"time"

	"meet-backend/internal/config"
	"meet-backend/internal/db"
	"meet-backend/internal/models"
	"meet-backend/internal/sfu"

	"github.com/google/uuid"
	"github.com/gorilla/websocket"
	"github.com/rs/zerolog/log"
)

const (
	// WebSocket tuning
	writeWait      = 10 * time.Second
	pongWait       = 60 * time.Second
	pingPeriod     = (pongWait * 9) / 10
	maxMessageSize = 64 * 1024 // 64KB per signaling message

	// Quality thresholds for P2P → SFU fallback
	maxPacketLoss = 8.0  // percent
	maxRTT        = 300.0 // ms
)

var upgrader = websocket.Upgrader{
	ReadBufferSize:  4096,
	WriteBufferSize: 4096,
	CheckOrigin: func(r *http.Request) bool {
		return true // CORS handled by middleware
	},
}

// Peer represents a connected WebSocket client.
type Peer struct {
	id       string
	userID   string
	username string
	roomID   string
	conn     *websocket.Conn
	send     chan []byte // buffered outbound channel
	mode     string     // "p2p" or "sfu"
	hub      *Hub
	mu       sync.Mutex
}

// Room holds all peers in a meeting room.
type Room struct {
	id    string
	peers map[string]*Peer // peerID → Peer
	mu    sync.RWMutex

	// SFU session (nil if all peers are in P2P mode)
	sfuSession *sfu.Session
}

// Hub manages all active rooms and peers.
type Hub struct {
	rooms  map[string]*Room
	mu     sync.RWMutex
	cfg    *config.Config
	db     *db.DB
	sfuMgr *sfu.Manager
}

// NewHub creates a new signaling hub.
func NewHub(cfg *config.Config, database *db.DB, sfuMgr *sfu.Manager) *Hub {
	return &Hub{
		rooms:  make(map[string]*Room),
		cfg:    cfg,
		db:     database,
		sfuMgr: sfuMgr,
	}
}

// ServeWS upgrades the HTTP connection to WebSocket and starts the peer loop.
func (h *Hub) ServeWS(w http.ResponseWriter, r *http.Request) {
	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		log.Error().Err(err).Msg("ws upgrade failed")
		return
	}

	peer := &Peer{
		id:   uuid.NewString(),
		conn: conn,
		send: make(chan []byte, 256),
		mode: "p2p",hub:  h,
	}

	go peer.writePump()
	go peer.readPump()
}

// --- Peer pumps ---

// readPump reads messages from the WebSocket connection.
// One goroutine per peer; all writes go through writePump.
func (p *Peer) readPump() {
	defer func() {
		p.hub.removePeer(p)
		p.conn.Close()
	}()

	p.conn.SetReadLimit(maxMessageSize)
	p.conn.SetReadDeadline(time.Now().Add(pongWait))
	p.conn.SetPongHandler(func(string) error {
		p.conn.SetReadDeadline(time.Now().Add(pongWait))
		return nil
	})

	for {
		_, raw, err := p.conn.ReadMessage()
		if err != nil {
			if websocket.IsUnexpectedCloseError(err, websocket.CloseGoingAway, websocket.CloseAbnormalClosure) {
				log.Warn().Str("peer", p.id).Err(err).Msg("ws read error")
			}
			return
		}
		p.hub.handleMessage(p, raw)
	}
}

// writePump drains the send channel and writes to the WebSocket.
func (p *Peer) writePump() {
	ticker := time.NewTicker(pingPeriod)
	defer func() {
		ticker.Stop()
		p.conn.Close()
	}()

	for {
		select {
		case msg, ok := <-p.send:
			p.conn.SetWriteDeadline(time.Now().Add(writeWait))
			if !ok {
				p.conn.WriteMessage(websocket.CloseMessage, []byte{})
				return
			}
			if err := p.conn.WriteMessage(websocket.TextMessage, msg); err != nil {
				return
			}
		case <-ticker.C:
			p.conn.SetWriteDeadline(time.Now().Add(writeWait))
			if err := p.conn.WriteMessage(websocket.PingMessage, nil); err != nil {
				return
			}
		}
	}
}

// send safely queues a message for the peer.
func (p *Peer) sendMsg(msg interface{}) {
	data, err := json.Marshal(msg)
	if err != nil {
		return
	}
	select {
	case p.send <- data:
	default:
		// Slow peer: drop message to avoid blocking
		log.Warn().Str("peer", p.id).Msg("send buffer full, dropping message")
	}
}

// --- Hub message routing ---

// handleMessage dispatches incoming messages by type.
func (h *Hub) handleMessage(p *Peer, raw []byte) {
	var msg models.SignalMessage
	if err := json.Unmarshal(raw, &msg); err != nil {
		log.Warn().Str("peer", p.id).Msg("invalid message")
		return
	}
	msg.From = p.id

	switch msg.Type {
	case "join":
		h.handleJoin(p, msg)
	case "leave":
		h.removePeer(p)
	case "offer", "answer", "ice-candidate":
		// P2P signaling: relay to target peer
		h.relay(p, msg)
	case "stats":
		// Peer reports its connection quality
		h.handleStats(p, msg)
	case "chat":
		// Broadcast chat message to room
		h.broadcast(p.roomID, msg, p.id)
	default:
		log.Warn().Str("type", msg.Type).Msg("unknown message type")
	}
}

// handleJoin processes a peer joining a room.
func (h *Hub) handleJoin(p *Peer, msg models.SignalMessage) {
	payload, err := json.Marshal(msg.Payload)
	if err != nil {
		return
	}
	var req models.JoinRequest
	if err := json.Unmarshal(payload, &req); err != nil {
		p.sendMsg(errMsg("invalid join payload"))
		return
	}

	// Validate room exists in DB
	var roomPass sql.NullString
	var maxPeers int
	err = h.db.QueryRow(
		`SELECT password, max_peers FROM rooms WHERE id = ? AND (expires_at IS NULL OR expires_at > ?)`,
		req.RoomID, time.Now().Unix(),
	).Scan(&roomPass, &maxPeers)
	if err != nil {
		p.sendMsg(errMsg("room not found or expired"))
		return
	}

	// Check room password if set
	if roomPass.Valid && roomPass.String != "" {
		if !checkPassword(req.Password, roomPass.String) {
			p.sendMsg(errMsg("wrong room password"))
			return
		}
	}

	// Get or create in-memory room
	h.mu.Lock()
	room, exists := h.rooms[req.RoomID]
	if !exists {
		room = &Room{
			id:    req.RoomID,
			peers: make(map[string]*Peer),
		}
		h.rooms[req.RoomID] = room
	}
	h.mu.Unlock()

	room.mu.Lock()
	// Enforce max peers
	if len(room.peers) >= maxPeers {
		room.mu.Unlock()
		p.sendMsg(errMsg("room is full"))
		return
	}
	room.peers[p.id] = p
	room.mu.Unlock()

	p.roomID = req.RoomID
	p.username = req.Username

	// Log join event
	go h.db.Exec(
		`INSERT INTO room_events (room_id, user_id, event, ts) VALUES (?, ?, 'join', ?)`,
		req.RoomID, p.userID, time.Now().Unix(),
	)

	// Send current peer list to the new peer
	peers := h.getPeerList(req.RoomID, p.id)
	p.sendMsg(models.SignalMessage{
		Type:    "joined",
		Payload: map[string]interface{}{"peers": peers, "your_id": p.id},
	})

	// Notify existing peers about the new peer
	h.broadcast(req.RoomID, models.SignalMessage{
		Type:    "peer-joined",
		From:    p.id,
		Payload: map[string]interface{}{"peer_id": p.id, "username": p.username},
	}, p.id)

	log.Info().Str("peer", p.id).Str("room", req.RoomID).Msg("peer joined")
}

// handleStats processes quality reports and triggers SFU fallback if needed.
func (h *Hub) handleStats(p *Peer, msg models.SignalMessage) {
	payload, err := json.Marshal(msg.Payload)
	if err != nil {
		return
	}
	var stats models.StatsReport
	if err := json.Unmarshal(payload, &stats); err != nil {
		return
	}

	// Determine if this peer should switch to SFU
	shouldFallback := stats.BitrateKbps < float64(h.cfg.SFUFallbackThresholdKbps) ||
		stats.PacketLossPct > maxPacketLoss ||
		stats.RTTMs > maxRTT

	if shouldFallback && p.mode == "p2p" {
		h.triggerSFUFallback(p)
	}
}

// triggerSFUFallback switches a peer from P2P to SFU mode.
func (h *Hub) triggerSFUFallback(p *Peer) {
	p.mu.Lock()
	if p.mode == "sfu" {
		p.mu.Unlock()
		return // already in SFU mode
	}
	p.mode = "sfu"
	p.mu.Unlock()

	log.Info().Str("peer", p.id).Str("room", p.roomID).Msg("switching peer to SFU mode")

	// Get or create SFU session for this room
	h.mu.RLock()
	room := h.rooms[p.roomID]
	h.mu.RUnlock()
	if room == nil {
		return
	}

	room.mu.Lock()
	if room.sfuSession == nil {
		room.sfuSession = h.sfuMgr.CreateSession(p.roomID)
	}
	session := room.sfuSession
	room.mu.Unlock()

	// Tell the peer to switch to SFU and provide TURN credentials
	turnCreds := h.sfuMgr.GetTURNCredentials(p.id)
	p.sendMsg(models.SignalMessage{
		Type: "sfu-switch",
		Payload: map[string]interface{}{
			"session_id":   session.ID,
			"turn_servers": turnCreds,
		},
	})
}

// relay forwards a signaling message to a specific peer (for P2P offer/answer/ICE).
func (h *Hub) relay(from *Peer, msg models.SignalMessage) {
	if msg.To == "" {
		return
	}
	h.mu.RLock()
	room := h.rooms[from.roomID]
	h.mu.RUnlock()
	if room == nil {
		return
	}

	room.mu.RLock()
	target := room.peers[msg.To]
	room.mu.RUnlock()

	if target != nil {
		target.sendMsg(msg)
	}
}

// broadcast sends a message to all peers in a room except the sender.
func (h *Hub) broadcast(roomID string, msg models.SignalMessage, exceptPeerID string) {
	h.mu.RLock()
	room := h.rooms[roomID]
	h.mu.RUnlock()
	if room == nil {
		return
	}

	room.mu.RLock()
	defer room.mu.RUnlock()

	for id, peer := range room.peers {
		if id != exceptPeerID {
			peer.sendMsg(msg)
		}
	}
}

// removePeer cleans up when a peer disconnects.
func (h *Hub) removePeer(p *Peer) {
	if p.roomID == "" {
		return
	}

	h.mu.RLock()
	room := h.rooms[p.roomID]
	h.mu.RUnlock()
	if room == nil {
		return
	}

	room.mu.Lock()
	delete(room.peers, p.id)
	empty := len(room.peers) == 0
	room.mu.Unlock()

	// Notify remaining peers
	h.broadcast(p.roomID, models.SignalMessage{
		Type:    "peer-left",
		From:    p.id,
		Payload: map[string]interface{}{"peer_id": p.id},
	}, p.id)

	// Clean up empty room from memory (DB record stays for history)
	if empty {
		h.mu.Lock()
		delete(h.rooms, p.roomID)
		h.mu.Unlock()
		log.Info().Str("room", p.roomID).Msg("room closed (empty)")
	}

	// Log leave event
	go h.db.Exec(
		`INSERT INTO room_events (room_id, user_id, event, ts) VALUES (?, ?, 'leave', ?)`,
		p.roomID, p.userID, time.Now().Unix(),
	)

	close(p.send)
	log.Info().Str("peer", p.id).Str("room", p.roomID).Msg("peer left")
}

// getPeerList returns info about all peers in a room except the given peer.
func (h *Hub) getPeerList(roomID, exceptID string) []models.PeerInfo {
	h.mu.RLock()
	room := h.rooms[roomID]
	h.mu.RUnlock()
	if room == nil {
		return nil
	}

	room.mu.RLock()
	defer room.mu.RUnlock()

	peers := make([]models.PeerInfo, 0, len(room.peers))
	for id, peer := range room.peers {
		if id != exceptID {
			peers = append(peers, models.PeerInfo{
				ID:       peer.id,
				Username: peer.username,
				Mode:     peer.mode,
			})
		}
	}
	return peers
}

// GetRoomStats returns live stats for a room (used by admin API).
func (h *Hub) GetRoomStats(roomID string) map[string]interface{} {
	h.mu.RLock()
	room := h.rooms[roomID]
	h.mu.RUnlock()
	if room == nil {
		return map[string]interface{}{"active": false}
	}

	room.mu.RLock()
	defer room.mu.RUnlock()

	p2pCount, sfuCount := 0, 0
	for _, peer := range room.peers {
		if peer.mode == "sfu" {
			sfuCount++
		} else {
			p2pCount++
		}
	}
	return map[string]interface{}{
		"active":     true,
		"peer_count": len(room.peers),
		"p2p_peers":  p2pCount,
		"sfu_peers":  sfuCount,
	}
}

// errMsg creates a standard error signal message.
func errMsg(reason string) models.SignalMessage {
	return models.SignalMessage{
		Type:    "error",
		Payload: map[string]string{"reason": reason},
	}
}

// Session represents an SFU session for a room.

type Session struct {

// sfuPeer holds the WebRTC peer connection for an SFU participant.
type sfuPeer struct {
	id   string
	pc   *webrtc.PeerConnection
	// Tracks published by this peer
	tracks []*webrtc.TrackLocalStaticRTP
}

// Manager manages all SFU sessions and the embedded TURN server.
type Manager struct {
	sessions map[string]*Session
	mu       sync.RWMutex
	cfg      *config.Config
	api      *webrtc.API
}

// NewManager creates a new SFU manager with a configured WebRTC API.
func NewManager(cfg *config.Config) *Manager {
	// Configure media engine with common codecs
	m := &webrtc.MediaEngine{}
	if err := m.RegisterDefaultCodecs(); err != nil {
		log.Fatal().Err(err).Msg("failed to register WebRTC codecs")
	}

	// Use a single interceptor registry for RTCP feedback (NACK, PLI, etc.)
	i := &interceptor.Registry{}
	if err := webrtc.RegisterDefaultInterceptors(m, i); err != nil {
		log.Fatal().Err(err).Msg("failed to register interceptors")
	}

	api := webrtc.NewAPI(
		webrtc.WithMediaEngine(m),
		webrtc.WithInterceptorRegistry(i),
	)

	return &Manager{
		sessions: make(map[string]*Session),
		cfg:      cfg,
		api:      api,
	}
}

// CreateSession creates or returns an existing SFU session for a room.
func (m *Manager) CreateSession(roomID string) *Session {
	m.mu.Lock()
	defer m.mu.Unlock()

	if s, ok := m.sessions[roomID]; ok {
		return s
	}
	s := &Session{
		ID:     uuid.NewString(),
		roomID: roomID,
		peers:  make(map[string]*sfuPeer),
	}
	m.sessions[roomID] = s
	log.Info().Str("room", roomID).Str("session", s.ID).Msg("SFU session created")
	return s
}

// CloseSession tears down an SFU session.
func (m *Manager) CloseSession(roomID string) {
	m.mu.Lock()
	s, ok := m.sessions[roomID]
	if ok {
		delete(m.sessions, roomID)
	}
	m.mu.Unlock()

	if s == nil {
		return
	}
	s.mu.Lock()
	defer s.mu.Unlock()
	for _, p := range s.peers {
		p.pc.Close()
	}
	log.Info().Str("room", roomID).Msg("SFU session closed")
}

// GetTURNCredentials generates time-limited TURN credentials using HMAC-SHA1.
// This is the standard RFC 8489 short-term credential mechanism.
func (m *Manager) GetTURNCredentials(peerID string) []map[string]interface{} {
	// Credentials expire in 1 hour
	expiry := time.Now().Add(time.Hour).Unix()
	username := fmt.Sprintf("%d:%s", expiry, peerID)

	mac := hmac.New(sha1.New, []byte(m.cfg.TURNSecret))
	mac.Write([]byte(username))
	password := base64.StdEncoding.EncodeToString(mac.Sum(nil))

	turnURL := fmt.Sprintf("turn:%s:%d", m.cfg.TURNPublicIP, m.cfg.TURNPort)
	return []map[string]interface{}{
		{
			"urls":       []string{turnURL},
			"username":   username,
			"credential": password,
		},
	}
}

type contextKey string

const (
	CtxUserID   contextKey = "user_id"
	CtxUsername contextKey = "username"
	CtxIsAdmin  contextKey = "is_admin"

// Logger logs each request with method, path, status, and duration.
func Logger(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		start := time.Now()
		rw := &responseWriter{ResponseWriter: w, status: 200}
		next.ServeHTTP(rw, r)
		log.Info().
			Str("method", r.Method).
			Str("path", r.URL.Path).
			Int("status", rw.status).
			Dur("duration", time.Since(start)).
			Msg("request")
	})
}

// CORS sets permissive CORS headers based on config.
func CORS(allowedOrigins string) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			w.Header().Set("Access-Control-Allow-Origin", allowedOrigins)
			w.Header().Set("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE,OPTIONS")
			w.Header().Set("Access-Control-Allow-Headers", "Content-Type,Authorization")
			if r.Method == http.MethodOptions {
				w.WriteHeader(http.StatusNoContent)
				return
			}
			next.ServeHTTP(w, r)
		})
	}
}

// Auth validates JWT and injects claims into context.
func Auth(cfg *config.Config) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			token := extractToken(r)
			if token == "" {
				http.Error(w, "unauthorized", http.StatusUnauthorized)
				return
			}
			claims, err := parseJWT(token, cfg.JWTSecret)
			if err != nil {
				http.Error(w, "invalid token", http.StatusUnauthorized)
				return
			}
			ctx := context.WithValue(r.Context(), CtxUserID, claims["sub"])
			ctx = context.WithValue(ctx, CtxUsername, claims["username"])
			ctx = context.WithValue(ctx, CtxIsAdmin, claims["is_admin"])
			next.ServeHTTP(w, r.WithContext(ctx))
		})
	}
}

// AdminOnly allows only requests with is_admin=true in JWT.
func AdminOnly(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		isAdmin, _ := r.Context().Value(CtxIsAdmin).(bool)
		if !isAdmin {
			http.Error(w, "forbidden", http.StatusForbidden)
			return
		}
		next.ServeHTTP(w, r)
	})
}

// --- helpers ---

func extractToken(r *http.Request) string {
	// Check Authorization header first
	if h := r.Header.Get("Authorization"); strings.HasPrefix(h, "Bearer ") {
		return strings.TrimPrefix(h, "Bearer ")
	}
	// Fallback: query param (useful for WebSocket upgrades)
	return r.URL.Query().Get("token")
}

func parseJWT(tokenStr, secret string) (jwt.MapClaims, error) {
	t, err := jwt.Parse(tokenStr, func(t *jwt.Token) (interface{}, error) {
		if _, ok := t.Method.(*jwt.SigningMethodHMAC); !ok {
			return nil, jwt.ErrSignatureInvalid
		}
		return []byte(secret), nil
	})
	if err != nil || !t.Valid {
		return nil, err
	}
	claims, ok := t.Claims.(jwt.MapClaims)
	if !ok {
		return nil, jwt.ErrTokenInvalidClaims
	}
	return claims, nil
}

type responseWriter struct {

	status int
}

func (rw *responseWriter) WriteHeader(code int) {
	rw.status = code
	rw.ResponseWriter.WriteHeader(code)
}

type loginRequest struct {
	Username string `json:"username"`
	Password string `json:"password"`
}

// Login handles both admin login and regular user login.
// POST /api/auth/login
func (h *AuthHandler) Login(w http.ResponseWriter, r *http.Request) {
	var req loginRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "bad request", http.StatusBadRequest)
		return
	}

	var userID string
	var isAdmin bool

	// Check admin credentials first (from env, not DB)
	if req.Username == h.cfg.AdminUsername && req.Password == h.cfg.AdminPassword {
		userID = "admin"
		isAdmin = true
	} else {
		// Regular user: look up in DB
		var hashedPass string
		err := h.db.QueryRow(
			`SELECT id, password FROM users WHERE username = ?`, req.Username,
		).Scan(&userID, &hashedPass)
		if err == sql.ErrNoRows {
			http.Error(w, "invalid credentials", http.StatusUnauthorized)
			return
		} else if err != nil {
			http.Error(w, "internal error", http.StatusInternalServerError)
			return
		}
		if err := bcrypt.CompareHashAndPassword([]byte(hashedPass), []byte(req.Password)); err != nil {
			http.Error(w, "invalid credentials", http.StatusUnauthorized)
			return
		}
	}

	token, err := h.issueJWT(userID, req.Username, isAdmin)
	if err != nil {
		http.Error(w, "token error", http.StatusInternalServerError)
		return
	}

	writeJSON(w, map[string]interface{}{
		"token":    token,
		"user_id":  userID,
		"username": req.Username,
		"is_admin": isAdmin,
	})
}

// Register creates a new user. Only allowed in public mode OR by admin.
// POST /api/auth/register
func (h *AuthHandler) Register(w http.ResponseWriter, r *http.Request) {
	// Check app mode
	var appMode string
	h.db.QueryRow(`SELECT app_mode FROM settings WHERE id = 1`).Scan(&appMode)

	// In private mode, only admin can register users (handled by AdminOnly middleware on route)
	var req loginRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "bad request", http.StatusBadRequest)
		return
	}
	if req.Username == "" || req.Password == "" {
		http.Error(w, "username and password required", http.StatusBadRequest)
		return
	}

	hash, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
	if err != nil {
		http.Error(w, "internal error", http.StatusInternalServerError)
		return
	}

	id := uuid.NewString()
	_, err = h.db.Exec(
		`INSERT INTO users (id, username, password, created_at) VALUES (?, ?, ?, ?)`,
		id, req.Username, string(hash), time.Now().Unix(),
	)
	if err != nil {
		http.Error(w, "username already taken", http.StatusConflict)
		return
	}

	w.WriteHeader(http.StatusCreated)
	writeJSON(w, map[string]string{"user_id": id, "username": req.Username})
}

func (h *AuthHandler) issueJWT(userID, username string, isAdmin bool) (string, error) {
	claims := jwt.MapClaims{
		"sub":      userID,
		"username": username,
		"is_admin": isAdmin,
		"exp":      time.Now().Add(24 * time.Hour).Unix(),"iat":      time.Now().Unix(),
	}
	return jwt.NewWithClaims(jwt.SigningMethodHS256, claims).SignedString([]byte(h.cfg.JWTSecret))
}

// CreateRoom creates a new meeting room.
// POST /api/rooms
func (h *RoomHandler) CreateRoom(w http.ResponseWriter, r *http.Request) {
	// Check if user is allowed to create rooms
	if err := h.checkCreatePermission(r); err != nil {
		http.Error(w, err.Error(), http.StatusForbidden)
		return
	}

	var req createRoomRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "bad request", http.StatusBadRequest)
		return
	}
	if strings.TrimSpace(req.Name) == "" {
		http.Error(w, "room name required", http.StatusBadRequest)
		return
	}
	if req.MaxPeers <= 0 {
		req.MaxPeers = 50
	}

	hostID, _ := r.Context().Value(middleware.CtxUserID).(string)

	var passHash string
	if req.Password != "" {
		h, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
		if err != nil {
			http.Error(w, "internal error", http.StatusInternalServerError)
			return
		}
		passHash = string(h)
	}

	var expiresAt *int64
	if req.ExpiresIn > 0 {
		t := time.Now().Add(time.Duration(req.ExpiresIn) * time.Second).Unix()
		expiresAt = &t
	}

	roomID := uuid.NewString()
	now := time.Now().Unix()

	_, err := h.db.Exec(
		`INSERT INTO rooms (id, name, host_id, password, max_peers, created_at, expires_at)
		 VALUES (?, ?, ?, ?, ?, ?, ?)`,
		roomID, req.Name, hostID, nullString(passHash), req.MaxPeers, now, nullInt64(expiresAt),
	)
	if err != nil {
		http.Error(w, "internal error", http.StatusInternalServerError)
		return
	}

	writeJSON(w, models.Room{
		ID:        roomID,
		Name:      req.Name,
		HostID:    hostID,
		HasPass:   passHash != "",
		MaxPeers:  req.MaxPeers,
		CreatedAt: now,
		ExpiresAt: expiresAt,
	})
}

	"meet-backend/internal/db"

	"meet-backend/internal/sfu"
	"meet-backend/internal/signaling"

func main() {
	// Pretty logging in dev, JSON in production
	cfg := config.Load()
	if cfg.Environment != "production" {
		log.Logger = log.Output(zerolog.ConsoleWriter{Out: os.Stderr})
	}

	// Open database
	database, err := db.Open(cfg.DBPath)
	if err != nil {
		log.Fatal().Err(err).Msg("failed to open database")
	}
	defer database.Close()

	// Initialize SFU manager
	sfuMgr := sfu.NewManager(cfg)

	// Initialize signaling hub
	hub := signaling.NewHub(cfg, database, sfuMgr)

	// Build handlers
	authH := handlers.NewAuthHandler(cfg, database)
	roomH := handlers.NewRoomHandler(cfg, database, hub)
	adminH := handlers.NewAdminHandler(database, hub)
	wsH := handlers.NewWSHandler(hub)

	// Router
	mux := http.NewServeMux()

	// Auth routes (public)
	mux.HandleFunc("/api/auth/login", authH.Login)

	// Register: public in public mode, admin-only in private mode
	// We handle the permission check inside the handler itself
	mux.Handle("/api/auth/register",
		middleware.Auth(cfg)(http.HandlerFunc(authH.Register)),
	)

	// Room routes (require auth)
	authMw := middleware.Auth(cfg)
	mux.Handle("/api/rooms", authMw(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		switch r.Method {
		case http.MethodGet:
			roomH.ListRooms(w, r)
		case http.MethodPost:
			roomH.CreateRoom(w, r)
		default:
			http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		}
	})))

	mux.Handle("/api/rooms/", authMw(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		switch r.Method {
		case http.MethodGet:
			roomH.GetRoom(w, r)
		case http.MethodDelete:
			roomH.DeleteRoom(w, r)
		default:
			http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		}
	})))

	// Admin routes (require auth + admin)
	adminMw := middleware.AdminOnly
	mux.Handle("/api/admin/settings",
		authMw(adminMw(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			switch r.Method {
			case http.MethodGet:
				adminH.GetSettings(w, r)
			case http.MethodPut:
				adminH.UpdateSettings(w, r)
			default:
				http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
			}
		}))))

	mux.Handle("/api/admin/users",
		authMw(adminMw(http.HandlerFunc(adminH.ListUsers))))

	mux.Handle("/api/admin/users/",
		authMw(adminMw(http.HandlerFunc(adminH.DeleteUser))))

	mux.Handle("/api/admin/rooms/",
		authMw(adminMw(http.HandlerFunc(adminH.GetRoomStats))))

	// WebSocket (auth via query param token)
	mux.Handle("/ws", authMw(http.HandlerFunc(wsH.ServeWS)))

	// Health check (no auth)
	mux.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(http.StatusOK)
		w.Write([]byte("ok"))
	})

	// Wrap with global middleware
	handler := middleware.Logger(middleware.CORS(cfg.AllowedOrigins)(mux))

	srv := &http.Server{
		Addr:         ":" + cfg.Port,
		Handler:      handler,
		ReadTimeout:  15 * time.Second,
		WriteTimeout: 15 * time.Second,
		IdleTimeout:  60 * time.Second,
	}

	// Graceful shutdown
	go func() {
		log.Info().Str("port", cfg.Port).Msg("server starting")
		if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			log.Fatal().Err(err).Msg("server error")
		}
	}()

	quit := make(chan os.Signal, 1)
	signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
	<-quit

	log.Info().Msg("shutting down...")
	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
	defer cancel()
	srv.Shutdown(ctx)
	log.Info().Msg("server stopped")
}
